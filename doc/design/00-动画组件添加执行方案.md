### Task 1: 创建 Reveal 组件（替代 AOS）

**文件**: `components/motion/Reveal.jsx`

```jsx
import { motion, useInView } from 'framer-motion'
import { useRef } from 'react'

export function Reveal({
  children,
  delay = 0,
  y = 24,
  duration = 0.6,
  className = '',
  once = true,
}) {
  const ref = useRef(null)
  const inView = useInView(ref, { once, amount: 0.2 })

  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, y }}
      animate={inView ? { opacity: 1, y: 0 } : {}}
      transition={{ duration, delay, ease: [0.22, 1, 0.36, 1] }}
      className={className}
    >
      {children}
    </motion.div>
  )
}
```

**使用示例** — 替换 `FeaturesBlocks.js` 中的静态卡片：

```jsx
// 之前：
<div className="p-6 bg-white rounded-md shadow-xl border">

// 之后：

<Reveal delay={index * 0.15}>
  <div className="p-6 bg-white rounded-md shadow-xl border ...">
```

### Task 2: 创建 StaggerList 组件

**文件**: `components/motion/StaggerList.jsx`

```jsx
import { motion, useInView } from 'framer-motion'
import { useRef } from 'react'

export function StaggerList({
  children,
  stagger = 0.12,
  className = '',
}) {
  const ref = useRef(null)
  const inView = useInView(ref, { once: true, amount: 0.1 })

  return (
    <motion.div
      ref={ref}
      initial="hidden"
      animate={inView ? 'show' : 'hidden'}
      variants={{
        hidden: {},
        show: { transition: { staggerChildren: stagger } },
      }}
      className={className}
    >
      {children}
    </motion.div>
  )
}

export function StaggerItem({ children, className = '' }) {
  return (
    <motion.div
      variants={{
        hidden: { opacity: 0, y: 20 },
        show: { opacity: 1, y: 0, transition: { duration: 0.5, ease: [0.22, 1, 0.36, 1] } },
      }}
      className={className}
    >
      {children}
    </motion.div>
  )
}
```

**使用** — `FeaturesBlocks.js` 的 6 张卡片：

```jsx
import { StaggerList, StaggerItem } from '@/components/motion/StaggerList'

<StaggerList stagger={0.12} className="grid gap-6 md:grid-cols-2 lg:grid-cols-3">
  {features.map((feature, index) => (
    <StaggerItem key={index}>
      {/* 原有卡片内容不变 */}
    </StaggerItem>
  ))}
</StaggerList>
```

### Task 3: 创建 ShimmerButton 组件

```jsx
import { motion } from 'framer-motion'

export function ShimmerButton({
  children,
  className = '',
  shimmerDuration = 4,
  ...props
}) {
  return (
    <motion.button
      whileHover={{ y: -2 }}
      whileTap={{ scale: 0.97 }}
      className={`relative overflow-hidden ${className}`}
      {...props}
    >
      {/* Shimmer 光效层 */}
      <motion.span
        className="absolute inset-0 pointer-events-none"
        animate={{ x: ['-120%', '140%'] }}
        transition={{
          duration: shimmerDuration,
          repeat: Infinity,
          ease: 'linear',
          repeatDelay: 0.8,
        }}
        style={{
          background:
            'linear-gradient(110deg, transparent 25%, rgba(255,255,255,0.2) 50%, transparent 75%)',
        }}
      />
      <span className="relative z-10">{children}</span>
    </motion.button>
  )
}
```

```jsx
import { motion } from 'framer-motion'

export function Marquee({
  children,
  speed = 50,
  direction = 'left',
  className = '',
}) {
  return (
    <div
      className={`overflow-hidden ${className}`}

        transition={{
          duration: speed,
          repeat: Infinity,
          ease: 'linear',
        }}
      >
        {children}
        {children}
      </motion.div>
    </div>
  )
}
```

### Task 8: 创建 SpotlightCard 组件

**文件**: `components/motion/SpotlightCard.jsx`

```jsx
import { useState } from 'react'
import { motion } from 'framer-motion'

export function SpotlightCard({ children, className = '' }) {
  const [position, setPosition] = useState({ x: 0, y: 0 })
  const [opacity, setOpacity] = useState(0)

  function handleMouseMove(e) {
    const rect = e.currentTarget.getBoundingClientRect()
    setPosition({
      x: e.clientX - rect.left,
      y: e.clientY - rect.top,
    })
  }

  return (
    <motion.div
      className={`relative overflow-hidden ${className}`}
      onMouseMove={handleMouseMove}
      onMouseEnter={() => setOpacity(1)}
      onMouseLeave={() => setOpacity(0)}
      whileHover={{ y: -4, transition: { duration: 0.3 } }}
    >
      {/* Spotlight 光斑 */}
      <motion.div
        className="absolute inset-0 pointer-events-none"
        animate={{ opacity }}
        transition={{ duration: 0.3 }}
        style={{
          background: `radial-gradient(300px circle at ${position.x}px ${position.y}px, rgba(59,130,246,0.1), transparent 60%)`,
        }}
      />
      {children}
    </motion.div>
  )
}
```

**使用** — 替换普通卡片 div：

```jsx
<SpotlightCard className="p-6 rounded-md shadow-xl border">
  {/* 原有卡片内容 */}
</SpotlightCard>
```

```jsx
import { motion } from 'framer-motion'

const orbs = [
  { x: '8%', y: '30%', size: 260, color: 'rgba(59,130,246,0.12)', dur: 14, delay: 0 },
  { x: '75%', y: '15%', size: 200, color: 'rgba(20,184,166,0.10)', dur: 18, delay: 1.5 },
  { x: '55%', y: '80%', size: 300, color: 'rgba(59,130,246,0.08)', dur: 20, delay: 0.8 },
]

export function FloatingOrbs() {
  return (
    <div className="absolute inset-0 -z-10 pointer-events-none overflow-hidden">
      {orbs.map((o, i) => (
        <motion.div
          key={i}
          className="absolute rounded-full"
          style={{
            left: o.x,
            top: o.y,
            width: o.size,
            height: o.size,
            background: `radial-gradient(circle, ${o.color} 0%, transparent 65%)`,
            filter: 'blur(8px)',
          }}
          animate={{
            x: [0, 25, -15, 0],
            y: [0, -20, 12, 0],
          }}
          transition={{
            duration: o.dur,
            delay: o.delay,
            repeat: Infinity,
            ease: 'easeInOut',
          }}
        />
      ))}
    </div>
  )
}
```

### Task 10: 创建 CountUp 组件

**文件**: `components/motion/CountUp.jsx`

```jsx
import { useEffect, useState } from 'react'

export function CountUp({ target, suffix = '', duration = 1400 }) {
  const [value, setValue] = useState(0)

  useEffect(() => {
    const start = performance.now()
    const tick = (now) => {
      const progress = Math.min((now - start) / duration, 1)
      const eased = 1 - Math.pow(1 - progress, 3)
      setValue(Math.round(target * eased))
      if (progress < 1) requestAnimationFrame(tick)
    }
    requestAnimationFrame(tick)
  }, [target, duration])

  return (
    <span>
      {value}
      {suffix}
    </span>
  )
}
```

**使用** — Testimonials 中的统计数字：

```jsx
<CountUp target={7000} suffix="+" />
```

## Phase 3：进阶效果（1-2 周）

### Task 11: MagneticButton 组件

```jsx
// components/motion/MagneticButton.jsx
import { motion } from 'framer-motion'
import { useState } from 'react'

export function MagneticButton({ children, className = '', strength = 8 }) {
  const [position, setPosition] = useState({ x: 0, y: 0 })

  function handleMouseMove(e) {
    const rect = e.currentTarget.getBoundingClientRect()
    const x = (e.clientX - rect.left - rect.width / 2) / rect.width
    const y = (e.clientY - rect.top - rect.height / 2) / rect.height
    setPosition({ x: x * strength, y: y * strength })
  }

  return (
    <motion.button
      className={className}
      onMouseMove={handleMouseMove}
      onMouseLeave={() => setPosition({ x: 0, y: 0 })}
      animate={position}
      transition={{ type: 'spring', stiffness: 200, damping: 15 }}
    >
      {children}
    </motion.button>
  )
}
```

### Task 12: BlurIn 文字动画

```jsx
// components/motion/BlurIn.jsx
import { motion, useInView } from 'framer-motion'
import { useRef } from 'react'

export function BlurIn({ children, delay = 0, className = '' }) {
  const ref = useRef(null)
  const inView = useInView(ref, { once: true, amount: 0.3 })

  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, filter: 'blur(8px)' }}
      animate={inView ? { opacity: 1, filter: 'blur(0px)' } : {}}
      transition={{ duration: 0.8, delay }}
      className={className}
    >
      {children}
    </motion.div>
  )
}
```

## 注意事项

1. **不要删除 AOS** — landing 主题仍在用 AOS，逐个替换即可，不要一次全删
2. **保持现有样式** — 只加动画层，不改布局和数据结构
3. **移动端降级** — framer-motion 动画默认支持 `useReducedMotion`，但要检查
4. **性能监控** — 每完成一个 Phase 跑一次 `yarn build`，确保 bundle 不会爆炸性增长
5. **深色模式** — 所有新组件需兼容 `dark:` 前缀
